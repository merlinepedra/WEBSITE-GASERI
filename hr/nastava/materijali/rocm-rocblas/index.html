               << ", " << ldb << ", " << ldc << std::endl;

                      beta,
                      m,
                      n,
                      k,
                      ha.data(),
                      a_stride_1,
                      a_stride_2,
                      hb.data(),
                      b_stride_1,

                      hc_gold.data(),

      max_relative_error

  }

  }

  CHECK_HIP_ERROR(hipFree(db));
  CHECK_HIP_ERROR(hipFree(dc));
  CHECK_ROCBLAS_ERROR(rocblas_destroy_handle(handle));
  return EXIT_SUCCESS;
}

                  int N,
                  int K,

                  int As2,
                  T*  B,
                  int Bs1,
                  int Bs2,
                  T*  C,
                  int Cs1,
                  int Cs2)

   {
     for(int i2 = 0; i2 < N; i2++)

       T t = 0.0;
       for(int i3 = 0; i3 < K; i3++)
       {
         t += A[i1 * As1 + i3 * As2] * B[i3 * Bs1 + i2 * Bs2];

     }
   }
